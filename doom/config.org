To install a package:

1. Declare them here in a =package!= statement,
2. Run =doom sync= in the shell,
3. Restart Emacs.

Use =C-h f package\!= to look up documentation for the =package!= macro.

To install =SOME-PACKAGE= from MELPA, ELPA or emacsmirror:

#+begin_src emacs-lisp :tangle no
(package! some-package)
#+end_src

To install a package directly from a remote git repo, you must specify a
=:recipe=. You'll find documentation on what =:recipe= accepts here:
https://github.com/radian-software/straight.el#the-recipe-format

#+begin_src emacs-lisp :tangle no
(package! another-package
  :recipe (:host github :repo "username/repo"))
#+end_src

If the package you are trying to install does not contain a =PACKAGENAME.el=
file, or is located in a subdirectory of the repo, you'll need to specify
=:files= in the =:recipe=:

#+begin_src emacs-lisp :tangle no
(package! this-package
  :recipe (:host github :repo "username/repo"
           :files ("some-file.el" "src/lisp/*.el")))
#+end_src

If you'd like to disable a package included with Doom, you can do so here with
the =:disable= property:

#+begin_src emacs-lisp :tangle no
(package! builtin-package :disable t)
#+end_src

You can override the recipe of a built in package without having to specify all
the properties for =:recipe=. These will inherit the rest of its recipe from Doom
or MELPA/ELPA/Emacsmirror:

#+begin_src emacs-lisp :tangle no
(package! builtin-package :recipe (:nonrecursive t))
(package! builtin-package-2 :recipe (:repo "myfork/package"))
#+end_src

Specify a =:branch= to install a package from a particular branch or tag. This is
required for some packages whose default branch isn't =master= (which our package
manager can't deal with; see radian-software/straight.el#279):

#+begin_src emacs-lisp :tangle no
(package! builtin-package :recipe (:branch "develop"))
#+end_src

Use =:pin= to specify a particular commit to install:

#+begin_src emacs-lisp :tangle no
(package! builtin-package :pin "1a2b3c4d5e")
#+end_src

Doom's packages are pinned to a specific commit and updated from release to
release. The =unpin!= macro allows you to unpin single packages...

#+begin_src emacs-lisp :tangle no
(unpin! pinned-package)
#+end_src

...or multiple packages:

#+begin_src emacs-lisp :tangle no
(unpin! pinned-package another-pinned-package)
#+end_src

...Or *all* packages (NOT RECOMMENDED; will likely break things):

#+begin_src emacs-lisp :tangle no
(unpin! t)
#+end_src

Custom Doom module for GitHub Copilot via =copilot.el=. Features smart tab

*STATUS:* You're not currently using Copilot. If you ever want AI-assisted

*** Config

#+begin_src emacs-lisp :tangle no
;; --- completion/copilot/config.el ---

(defun private/org-at-heading-start-p ()
  "Return t if point is at the beginning of a heading."
  (when (org-at-heading-p)
    (let ((pos (point)))
      (save-excursion
        (org-end-of-line)
        (org-beginning-of-line)
        (equal pos (point))))))

(defun private/copilot-indent-or-complete (&optional arg)
  "Accept completion from copilot or fallback to <tab> behavior."
  (interactive "P")
  (cond
   ((use-region-p)
    (indent-for-tab-command))
   ((and (derived-mode-p 'org-mode) (eq major-mode 'org-mode))
    (or (private/org-at-heading-start-p) (org-at-drawer-p)) (org-cycle))
   (t
    (let ((old-point (point))
          (old-tick (buffer-chars-modified-tick))
          (tab-always-indent t))
      (unless (or (and (not (copilot--overlay-visible)) (eolp))
                  (and (copilot--overlay-visible)
                       (eq (eol) (overlay-start copilot--overlay))))
        (indent-for-tab-command))
      (when (and (eq old-point (point))
                 (eq old-tick (buffer-chars-modified-tick)))
        (if (copilot--overlay-visible)
            (copilot-accept-completion)
          (copilot-complete)))))))

(defun private/toggle-company-copilot ()
  "Toggle between copilot-mode, company-mode, or neither."
  (interactive)
  (cond ((bound-and-true-p copilot-mode)
         (copilot-mode -1)
         (company-mode 1)
         (message "company-mode"))
        ((bound-and-true-p company-mode)
         (company-mode -1)
         (message "completions off"))
        (t
         (copilot-mode 1)
         (message "copilot-mode"))))

(use-package! copilot
  :hook (prog-mode . copilot-mode)
  :config
  (map! :map copilot-mode-map "<tab>"
        #'private/copilot-indent-or-complete)
  (map! :leader :desc "Completion mode" :prefix "t" "C"
        #'private/toggle-company-copilot)

  (if (modulep! :lang web)
      (defadvice! +web/indent-or-yas-or-emmet-expand-a ()
        :before-while #'+web/indent-or-yas-or-emmet-expand
        (if (and (bound-and-true-p copilot-mode)
                 (copilot--overlay-visible))
            (progn
              (message "copilot on and overlay visible")
              (private/copilot-indent-or-complete)
              nil)
          t)))

  (add-hook 'copilot-mode-hook (lambda ()
                                 (company-mode -1)
                                 (yas-minor-mode -1))))
#+end_src

** Auto theme :archive:

Automatic light/dark theme switching with three strategies: fixed light/dark,
solar-based (sunrise/sunset from =calendar=), or macOS appearance hooks
(emacs-mac's =mac-effective-appearance-change-hook=). Falls back from macOS →
solar if not running the emacs-mac port.

Doom has no built-in auto theme switching --- =doom-theme= is a single static
theme. This would replace that with dynamic switching.

*** Configuration

Customizable light/dark themes and switching mode. Defaults to =modus-operandi=
/ =modus-vivendi= with macOS appearance tracking.

#+begin_src emacs-lisp :tangle no
(defcustom my/autotheme-light-theme 'modus-operandi
  "Preferred light theme."
  :type 'symbol :group 'my-autotheme)

(defcustom my/autotheme-dark-theme 'modus-vivendi
  "Preferred dark theme."
  :type 'symbol :group 'my-autotheme)

(defcustom my/autotheme-mode 'macos
  "Theme switching mode.
`light' or `dark' for fixed theme. `native' for solar-based switching.
`macos' for macOS appearance hooks (falls back to native)."
  :type '(choice (const light) (const dark) (const native) (const macos))
  :group 'my-autotheme)
#+end_src

*** Solar detection

Uses Emacs's built-in =solar= library to compute sunrise/sunset for the current
location (set via =calendar-latitude= / =calendar-longitude=). Returns whether
it's currently daytime.

#+begin_src emacs-lisp :tangle no
(defun my/autotheme--sunrise-sunset ()
  "Return (sunrise sunset) in minutes for today."
  (require 'solar)
  (let ((hours (solar-sunrise-sunset (calendar-current-date))))
    (mapcar (lambda (s) (* 60 s))
            (list (caar hours) (caadr hours)))))

(defun my/autotheme--now-minutes ()
  "Return current time in minutes since midnight."
  (let ((ct (decode-time)))
    (+ (* 60 (decoded-time-hour ct))
       (decoded-time-minute ct))))

(defun my/autotheme--day-p ()
  "Return t if the sun has risen."
  (let ((ss (my/autotheme--sunrise-sunset)))
    (<= (car ss) (my/autotheme--now-minutes) (cadr ss))))
#+end_src

*** System detection and theme resolution

Detects whether emacs-mac is available for native appearance hooks. Resolves
the effective mode and returns =dark= or =light= accordingly.

#+begin_src emacs-lisp :tangle no
(defun my/autotheme--emacs-mac-p ()
  "Return t if using railwaycat's emacs-mac build."
  (and (eq system-type 'darwin) (featurep 'mac-win)))

(defun my/autotheme--effective-mode ()
  "Return the effective mode after system detection."
  (if (and (eq my/autotheme-mode 'macos) (not (my/autotheme--emacs-mac-p)))
      'native
    my/autotheme-mode))

(defun my/autotheme--dark-or-light ()
  "Return `dark' or `light' based on mode."
  (let ((mode (my/autotheme--effective-mode)))
    (cond ((memq mode '(dark light)) mode)
          ((eq mode 'macos)
           (let ((appearance (plist-get (mac-application-state) :appearance)))
             (if (string= appearance "NSAppearanceNameDarkAqua") 'dark 'light)))
          ((eq mode 'native)
           (if (my/autotheme--day-p) 'light 'dark)))))

(defun my/autotheme--preferred-theme ()
  "Return the preferred theme."
  (if (eq (my/autotheme--dark-or-light) 'light)
      my/autotheme-light-theme
    my/autotheme-dark-theme))
#+end_src

*** Theme loading and toggle

Loads a theme with an optional dissolve animation (emacs-mac). Only switches if
the theme actually changed. =my/autotheme-toggle= manually flips between light
and dark.

#+begin_src emacs-lisp :tangle no
(defun my/autotheme-load (&optional theme)
  "Load THEME or the preferred theme. Animate on macOS if available."
  (let ((theme (or theme (my/autotheme--preferred-theme))))
    (unless (equal (car custom-enabled-themes) theme)
      (when (fboundp 'mac-start-animation)
        (mac-start-animation nil :type 'dissolve :duration 0.5))
      (mapc #'disable-theme custom-enabled-themes)
      (load-theme theme t))))

(defun my/autotheme-toggle ()
  "Toggle between light and dark themes."
  (interactive)
  (my/autotheme-load
   (if (equal (car custom-enabled-themes) my/autotheme-dark-theme)
       my/autotheme-light-theme
     my/autotheme-dark-theme)))
#+end_src

*** Initialization

Call =my/autotheme-init= after startup. On macOS with emacs-mac, hooks into
=mac-effective-appearance-change-hook= for instant switching when System
Preferences changes. On other systems, polls every 5 minutes using solar time.

#+begin_src emacs-lisp :tangle no
(defvar my/autotheme--timer nil)

(defun my/autotheme-init ()
  "Load preferred theme and set up automatic switching."
  (my/autotheme-load)
  (let ((mode (my/autotheme--effective-mode)))
    (cond ((eq mode 'native)
           (setq my/autotheme--timer
                 (run-with-timer 300 300 #'my/autotheme-load)))
          ((eq mode 'macos)
           (add-hook 'mac-effective-appearance-change-hook
                     #'my/autotheme-load)))))
#+end_src

*** Disable scroll interpolation in TTY

Doom's =smooth-scroll +interpolate= module advises =scroll-up= / =scroll-down= to
route through =good-scroll-move= for animated scrolling. But =good-scroll-mode= is
global --- toggling it per-frame isn't possible. In a terminal, the animation
just causes flickering and lag.

Fix: remove Doom's advice and replace it with versions that also check
=display-graphic-p=. In TTY frames the original =scroll-up= / =scroll-down= are
called directly.

#+begin_src emacs-lisp :tangle no
(after! good-scroll
  (advice-remove #'scroll-up #'good-scroll--scroll-up)
  (advice-remove #'scroll-down #'good-scroll--scroll-down)
  (defadvice! +good-scroll-up-gui-only-a (fn &optional arg)
    :around #'scroll-up
    (if (and good-scroll-mode (display-graphic-p))
        (good-scroll-move (good-scroll--convert-line-to-step arg))
      (funcall fn arg)))
  (defadvice! +good-scroll-down-gui-only-a (fn &optional arg)
    :around #'scroll-down
    (if (and good-scroll-mode (display-graphic-p))
        (good-scroll-move (- (good-scroll--convert-line-to-step arg)))
      (funcall fn arg))))
#+end_src

;; -- Native (solar-based) mode --

(defun +autotheme--sunrise-sunset ()
  "Return time in minutes of sunrise and sunset for today."
  (require 'solar)
  (let* ((sunrise-sunset-hours (solar-sunrise-sunset
                                (calendar-current-date))))
    (mapcar (lambda (s) (* 60 s))
            (list (caar sunrise-sunset-hours)
                  (caadr sunrise-sunset-hours)))))

(defun +autotheme--now ()
  "Return current time in minutes."
  (let* ((cts (current-time-string)))
    (+ (* 60 (string-to-number (substring cts 11 13)))
       (string-to-number (substring cts 14 16))
       (/ (string-to-number (substring cts 17 19)) 60))))

(defun +autotheme--day-p ()
  "Return t if the sun has risen."
  (require 'solar)
  (let* ((sunrise-sunset-minutes (+autotheme--sunrise-sunset)))
    (<= (* 60 (car sunrise-sunset-minutes))
        (+autotheme--now)
        (* 60 (cadr sunrise-sunset-minutes)))))

(defun +autotheme--next-transition ()
  "Return next theme transition time in minutes."
  (let* ((sunrise-sunset-minutes (+autotheme--sunrise-sunset)))
    (if (+autotheme--day-p)
        (cadr sunrise-sunset-minutes)
      (car sunrise-sunset-minutes))))

(defvar +autotheme--timer nil)

(defun +autotheme--start-timer (&optional no-update)
  "Update theme unless NO-UPDATE. Set timer for next transition."
  (let ((timer-time
         (round (* 60 (- (+autotheme--now)
                         (+autotheme--next-transition)))))))
  (unless no-update (+autotheme-load-theme))
  (setq +autotheme--timer
        (run-with-timer timer-time nil
                        #'autotheme--start-timer)))

;; -- macOS mode --

(defun +autotheme--emacs-mac-p ()
  "Return t if using railwaycat's emacs-mac build."
  (and IS-MAC (featurep 'mac-win)))

;; -- Common --

(defun +autotheme--mode ()
  "Return autotheme-mode after accounting for system."
  (let ((mode +autotheme-mode))
    (cond ((eq mode 'native) 'native)
          ((eq mode 'macos)
           (or (and (+autotheme--emacs-mac-p) 'macos) 'native))
          (t mode))))

(defun +autotheme--dark-or-light ()
  "Return dark or light depending on mode/time."
  (let ((mode (+autotheme--mode)))
    (cond ((member mode '(dark light)) mode)
          ((eq mode 'macos)
           (let ((appearance (plist-get (mac-application-state)
                                       :appearance)))
             (cond ((string= appearance "NSAppearanceNameDarkAqua") 'dark)
                   ((string= appearance "NSAppearanceNameAqua") 'light)
                   (t nil))))
          ((eq mode 'native)
           (if (+autotheme--day-p) 'light 'dark))
          (t nil))))

(defun +autotheme--preferred-theme ()
  "Return preferred theme according to autotheme-mode."
  (let ((cat (+autotheme--dark-or-light)))
    (cond ((eq cat 'light) +autotheme-light-theme)
          ((eq cat 'dark) +autotheme-dark-theme)
          (t doom-theme))))

(defun +autotheme-load-theme (&optional theme force-reload no-animation)
  "Set doom-theme to THEME and load.
If NO-ANIMATION, skip dissolve. If FORCE-RELOAD, reload even if unchanged."
  (let ((theme (or theme (+autotheme--preferred-theme))))
    (unless (and (eq doom-theme theme) (not force-reload))
      (setq doom-theme theme)
      (unless no-animation
        (mac-start-animation nil :type 'dissolve :duration 0.5))
      (load-theme doom-theme t))))

(defun +autotheme/toggle (&rest args)
  "Toggle between light and dark themes."
  (interactive)
  (apply #'+autotheme-load-theme
         (if (eq doom-theme +autotheme-dark-theme)
             +autotheme-light-theme
           +autotheme-dark-theme) args))

(defun +autotheme-init ()
  "Load preferred theme, start timer or add hooks for auto switching."
  (+autotheme-load-theme)
  (let ((auto-mode (+autotheme--mode)))
    (cond ((eq auto-mode 'native)
           (+autotheme--start-timer t))
          ((eq auto-mode 'macos)
           (add-hook! 'mac-effective-appearance-change-hook
             (+autotheme-load-theme))))))

*** Stale buffer detection and marking

Marks buffers in ibuffer that are either:
- *Old*: not viewed in 6+ hours (configurable via =my/buffer-old-time=)
- *Dissociated*: visiting a file or directory that no longer exists

Bound to =* t= in ibuffer. With prefix arg, marks all saved buffers regardless
of age.

#+begin_src emacs-lisp :tangle no
(defvar my/buffer-old-time 6
  "Number of hours before a buffer is considered old.")

(defun my/buffer-old-p (buf)
  "Return non-nil if BUF hasn't been viewed in `my/buffer-old-time' hours."
  (with-current-buffer buf
    (when buffer-display-time
      (time-less-p
       (* 60 60 my/buffer-old-time)
       (time-since buffer-display-time)))))

(defun my/buffer-dissociated-p (buf)
  "Return non-nil if BUF visits a file/dir that no longer exists."
  (with-current-buffer buf
    (or
     (and buffer-file-name
          (not (file-exists-p buffer-file-name)))
     (and (eq major-mode 'dired-mode)
          (boundp 'dired-directory)
          (stringp dired-directory)
          (not (file-exists-p (file-name-directory dired-directory)))))))

(defun my/ibuffer-mark-stale-buffers (arg)
  "Mark old unmodified buffers and buffers visiting deleted files.
With prefix ARG, mark all saved buffers regardless of age."
  (interactive "P")
  (ibuffer-mark-on-buffer
   (lambda (buf) (or (my/buffer-dissociated-p buf)
                     (and (buffer-file-name buf)
                          (not (buffer-modified-p buf))
                          (if arg t (my/buffer-old-p buf)))))))

(after! ibuffer
  (define-key ibuffer-mode-map "* t" #'my/ibuffer-mark-stale-buffers))

*** Key lookup

Searches all loaded keymaps to find where a key sequence is bound. Press any
key combo and see every keymap that defines it. Useful for debugging binding
conflicts. Doom's =describe-key= (=C-h k=) shows the active binding but not
all keymaps that define it.

#+begin_src emacs-lisp :tangle no
(defun my/key-lookup (key-sequence)
  "Find all keymaps where KEY-SEQUENCE is bound."
  (interactive
   (list (read-key-sequence "Press key: ")))
  (let ((result))
    (mapatoms
     (lambda (ob)
       (when (and (boundp ob) (keymapp (symbol-value ob)))
         (let ((m (lookup-key (symbol-value ob) key-sequence)))
           (when (and m (or (functionp m)
                            (symbolp m)
                            (keymapp m)))
             (push ob result)))))
     obarray)
    (message "Key sequence %s is bound in: %s"
             (key-description key-sequence)
             (if result
                 (mapconcat #'symbol-name result ", ")
               "nothing"))))
#+end_src

  "Flymake backend that runs `org-lint' on the current buffer."
  (let ((diags '()))
    (dolist (entry (org-lint))
      (let* ((vec (cadr entry))           ; [LINE TRUST DESCRIPTION CHECKER]
             (line-str (aref vec 0))
             (trust (aref vec 1))
             (msg (aref vec 2))

                                       (car region) (cdr region)
                                       type msg)
              diags)))

*** Formatting

Normalize blank lines and whitespace in org buffers on save. Rules:

- One blank line before headings (two before top-level =*= headings)
- One blank line after a heading line
- One blank line before/after blocks (=#+begin=​/​=#+end=)
- Collapse runs of 3+ blank lines to 2
- Strip trailing whitespace
- Single newline at EOF

#+begin_src emacs-lisp
(defun +org/format-buffer ()
  "Normalize whitespace in the current org buffer."
  (interactive)
  (when (derived-mode-p 'org-mode)
    (let ((inhibit-modification-hooks t)
          (save-silently t))
      (save-excursion
        (save-restriction
          (widen)
          ;; Strip trailing whitespace
          (delete-trailing-whitespace)
          ;; Collapse 3+ consecutive blank lines to 2
          (goto-char (point-min))
          (while (re-search-forward "\n\\{4,\\}" nil t)
            (replace-match "\n\n\n"))
          ;; Blank line before blocks (#+begin_...)
          (goto-char (point-min))
          (while (re-search-forward
                  "\\([^\n]\\)\n\\(#\\+begin_\\)" nil t)
            (replace-match "\\1\n\n\\2"))
          ;; Blank line after blocks (#+end_...)
          (goto-char (point-min))
          (while (re-search-forward
                  "\\(#\\+end_[^\n]*\\)\n\\([^\n#]\\)" nil t)
            (replace-match "\\1\n\n\\2"))
          ;; Two blank lines before top-level headings
          (goto-char (point-min))
          (while (re-search-forward "\\([^\n]\\)\n\\(\\* [^\n]\\)" nil t)
            (replace-match "\\1\n\n\n\\2"))
          ;; One blank line before deeper headings
          (goto-char (point-min))
          (while (re-search-forward
                  "\\([^\n]\\)\n\\(\\*\\{2,\\} [^\n]\\)" nil t)
            (replace-match "\\1\n\n\\2"))
          ;; One blank line after heading lines (but not before
          ;; :PROPERTIES: drawers or planning lines which must follow
          ;; headings immediately)
          (goto-char (point-min))
          (while (re-search-forward
                  "\\(^\\*+ [^\n]*\\)\n\\([^\n*:]\\)" nil t)
            (unless (save-excursion
                      (goto-char (match-beginning 2))
                      (looking-at "\\(DEADLINE\\|SCHEDULED\\|CLOSED\\)"))
              (replace-match "\\1\n\n\\2")))
          ;; Fix list indentation (org-indent-region mishandles
          ;; property drawers, so only apply to list items)
          (goto-char (point-min))
          (while (re-search-forward "^[ \t]*[-+*] \\|^[ \t]*[0-9]+[.)]] " nil t)
            (org-indent-region
             (line-beginning-position)
             (save-excursion (org-end-of-item-list) (point))))
          ;; Single newline at EOF
          (goto-char (point-max))
          (delete-blank-lines)
          (unless (bolp) (insert "\n")))))))

(add-hook 'org-mode-hook
          (lambda ()
            (add-hook 'before-save-hook #'+org/format-buffer nil t)))
#+end_src

**** Org: project detection

Checks if an org heading is a "project" (has TODO subtasks). Useful for custom
agenda views that separate projects from standalone tasks. From Bernt Hansen's
GTD setup.

#+begin_src emacs-lisp :tangle no
(defun +org-entry-is-project-p ()
  "Return non-nil if current heading has TODO subtasks."
  (save-restriction
    (widen)
    (let ((has-subtask)
          (subtree-end (save-excursion (org-end-of-subtree t)))
          (is-a-task (member (nth 2 (org-heading-components))
                             org-todo-keywords-1)))
      (save-excursion
        (forward-line 1)
        (while (and (not has-subtask)
                    (< (point) subtree-end)
                    (re-search-forward "^\*+ " subtree-end t))
          (when (member (org-get-todo-state) org-todo-keywords-1)
            (setq has-subtask t))))
      (and is-a-task has-subtask))))
#+end_src

** Modular config loader :doom:archive:

A custom Doom module that auto-discovers and loads config files matching Doom
module names from =$DOOMDIR/config/=. For example, if you have Doom's =:tools
magit= enabled, it loads =$DOOMDIR/config/tools/magit.el= if it exists. Includes
load time tracking and benchmark display.

*STATUS:* This is what powered the modular config structure (all those
=config/ui/config.el=, =config/app/slack.el= files). Now that everything is

(defgroup modular nil
  "Modular config"
  :group 'doom)

  "Non-volatile local storage for private config files."
  :type 'string)

  :group 'modular
  :type 'string)

  "Directory for private config .el files."
  :group 'modular
  :type 'string)

(defcustom +modular-show-load-times t
  "When t, show time per file. When `overview', show total only."
  :group 'modular
  :type 'boolean)

(defcustom +modular-enabled t
  "When non-nil, load modular config files."
  :group 'modular
  :type 'boolean)

(defvar +modular-after-init-hook nil)

;; Show load times after Doom's benchmark
(defadvice! +modular--show-load-time ()
  :after 'doom-display-benchmark-h
  (when (and +modular-enabled +modular-show-load-times)
    (dolist (kv +modular--load-times)
      (let ((file (car kv))
            (time (cdr kv)))
        (when (or (eq file t)
                  (not (eq +modular-show-load-times 'overview)))
          (let* ((file (if (stringp file)
                           (file-relative-name (doom-path file)
                                               doom-user-dir)
                         "modular config"))
                 (time (or time +modular--load-time))
                 (msg (format "Spent %.3fs in %s." time file)))
            (if (not (numberp time))
                (message "Unable to display load time!")
              (message msg))))))))

;; Load on init
(add-hook! 'doom-after-init-hook
  (when +modular-enabled
    (setq t0 (current-time))
    (+modular/load)
    (doom-run-hooks '+modular-after-init-hook)
    (let ((load-time (float-time (time-subtract (current-time) t0))))
      (+modular/add-load-time t load-time))
    (makunbound 't0)))

*** Autoloads (file discovery and loading)

#+begin_src emacs-lisp :tangle no
;; --- modules/config/modular/autoload.el ---

(defun +modular--file-path (category &optional module file)
  (let* ((path (doom-path
                +modular-config-dir
                (string-replace ":" "" (symbol-name category))
                (when module (symbol-name module)))))
    (if file
        (concat (expand-file-name file path) ".el")
      (concat path ".el"))))

(defun modular--module-files (category module)
  "Return list of filenames to look for given MODULE in CATEGORY."
  (let* ((files (mapcar (lambda (args)
                          (apply #'+modular--file-path args))
                        `((,category)
                          (,category ,module)
                          (,category nil "config")
                          (,category ,module "config")))))
    (append files (mapcar (lambda (flag)
                            (+modular--file-path category module
                                                 (symbol-name flag)))
                          (doom-module-get category module :flags)))))

(defun +modular/files ()
  "Return list of all files which would be loaded."
  (let ((module-keys (doom-module-list))
        (module-paths-func
         (lambda (m) (modular--module-files (car m) (cdr m)))))
    (seq-filter #'file-exists-p
                (seq-uniq
                 (apply #'append
                        (mapcar module-paths-func module-keys))))))

(defun +modular/add-load-time (key value)
  (appendq! +modular--load-times (list (cons key value))))

(defun +modular/load-file (file)
  (setq t1 (current-time))
  (with-demoted-errors
      (concat "Error loading " file " from modular config: %S")
    (load file))
  (let ((load-time (float-time (time-subtract (current-time) t1))))
    (+modular/add-load-time file load-time))
  (makunbound 't1))

(defun +modular/load ()
  "For each Doom module, load corresponding file in $DOOMDIR/config."
  (seq-do #'+modular/load-file (+modular/files)))
#+end_src
